/* 
Items:
    1. title
    2. description
    3. description icon
    4. add new item
    5. new item icon
    6. remove item
    7. remove item icon
*/
const list_background = document.querySelector('.list__background');
const list_item = document.createElement('div') //('.list-item');
const form = document.createElement('form') //('.form-title');
const textInput = document.createElement('input') //('.form-title__input');
const todo_list = document.querySelector('.todo-list');
const todoHistory = [];

form.addEventListener('submit', (event) => {
  event.preventDefault();
  const text = textInput.value;
  if (text.length > 0) {
    const todo = document.createElement('li');
    todo.setAttribute('margin', '10px;')
    todo.classList.add('todo');
    todo.textContent = text;
    todo/*remove-icon*/.addEventListener('click', () => {
      todo.remove();
    });
    todoHistory.push(todo);
    textInput.value = '';
    textInput.focus();

    list_background.appendChild(todo_list);
    todo_list.appendChild(todo);
  }
});
